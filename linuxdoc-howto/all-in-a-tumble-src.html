234/* parse-SNIP: rst_mode */
235/**
236 * rst_mode - dummy to demonstrate reST & kernel-doc markup in comments
237 * @a: first argument
238 * @b: second argument
239 * Context: :c:func:`in_gizmo_mode`.
240 *
241 * Long description. This function has two integer arguments. The first is
242 * ``parameter_a`` and the second is ``parameter_b``.
243 *
244 * As long as the reST / sphinx-doc toolchain uses `intersphinx
245 * <http://www.sphinx-doc.org/en/stable/ext/intersphinx.html>`__ you can refer
246 * definitions *outside* like :c:type:`struct media_device <media_device>`.  If
247 * the description of ``media_device`` struct is found in any of the intersphinx
248 * locations, a hyperref to this target is generated a build time.
249 *
250 * Example:
251 *   int main() {
252 *     printf("Hello World\n");
253 *     return 0;
254 *   }
255 *
256 * Return: Sum of ``parameter_a`` and the second is ``parameter_b``.
257 *
258 * highlighting:
259 * The highlight pattern, are non regular reST markups. They are only available
260 * within kernel-doc comments, helping C developers to write short and compact
261 * documentation.
262 *
263 * - user_function() : function
264 * - @a : name of a parameter
265 * - &struct my_struct : name of a structure (including the word struct)
266 * - &union my_union : name of a union
267 * - &my_struct->a or &my_struct.b -  member of a struct or union.
268 * - &enum my_enum : name of a enum
269 * - &typedef my_typedef : name of a typedef
270 * - %CONST : name of a constant.
271 * - $ENVVAR : environmental variable
272 *
273 * The kernel-doc parser translates the pattern above to the corresponding reST
274 * markups. You don't have to use the *highlight* pattern, if you prefer *pure*
275 * reST, use the reST markup.
276 *
277 * - :c:func:`user_function` : function
278 * - ``a`` : name of a parameter
279 * - :c:type:`struct my_struct <my_struct>` : name of a structure (including the word struct)
280 * - :c:type:`union my_union <my_union>` : name of a union
281 * - :c:type:`my_struct->a <my_struct>` or :c:type:`my_struct.b <my_struct>` -  member of a struct or union.
282 * - :c:type:`enum my_enum <my_enum>` : name of a enum
283 * - :c:type:`typedef my_typedef <my_typedef>` : name of a typedef
284 * - ``CONST`` : name of a constant.
285 * - ``\$ENVVAR`` : environmental variable
286 *
287 * Since the prefixes ``$...``, ``&...`` and ``@...`` are used to markup the
288 * highlight pattern, you have to escape them in other uses: \$lorem, \&lorem,
289 * \%lorem and \@lorem. To esacpe from function highlighting, use lorem\().
290 *
291 * Parser Mode:
292 * This is an example with activated reST additions, in this section you will
293 * find some common inline markups.
294 *
295 * Within the *reST mode* the kernel-doc parser pass through all markups to the
296 * reST toolchain, except the *vintage highlighting* but including any
297 * whitespace. With this, the full reST markup is available in the comments.
298 *
299 * This is a link to the `Linux kernel source tree
300 * <https://git.kernel.org/cgit/linux/kernel/git/torvalds/linux.git/>`_.
301 *
302 * This description is only to show some reST inline markups like *emphasise*
303 * and **emphasis strong**. The following is a demo of a reST list markup:
304 *
305 * Definition list:
306 * :def1: lorem
307 * :def2: ipsum
308 *
309 * Ordered List:
310 * - item one
311 * - item two
312 * - item three with
313 *   a linebreak
314 *
315 * Literal blocks:
316 * The next example shows a literal block::
317 *
318 *     +------+          +------+
319 *     |\     |\        /|     /|
320 *     | +----+-+      +-+----+ |
321 *     | |    | |      | |    | |
322 *     +-+----+ |      | +----+-+
323 *      \|     \|      |/     |/
324 *       +------+      +------+
325 *        foo()         bar()
326 * 
327 * Highlighted code blocks:
328 * The next example shows a code block, with highlighting C syntax in the
329 * output.
330 *
331 * .. code-block:: c
332 *
333 *     // Hello World program
334 *     #include<stdio.h>
335 *     int main()
336 *     {
337 *        printf("Hello World");
338 *     }
339 *
340 *
341 * reST sectioning:
342 *
343 * colon markup: sectioning by colon markup in reST mode is less ugly. ;-)
344 *
345 * A kernel-doc section like *this* section is translated into a reST
346 * *subsection*. This means, you can only use the following *sub-levels* within a
347 * kernel-doc section.
348 *
349 * a subsubsection
350 * ^^^^^^^^^^^^^^^
351 *
352 * lorem ipsum
353 *
354 * a paragraph
355 * """""""""""
356 *
357 * lorem ipsum
358 *
359 */
360int rst_mode(int a, char *b)
361{
362  return a + b;
363}

369/**
370 * vintage - short description of this function
371 * @parameter_a: first argument
372 * @parameter_b: second argument
373 * Context: in_gizmo_mode().
374 *
375 * This is a test of a typical markup from *vintage* kernel-doc.  Don't look to
376 * close here, it is only for testing some kernel-doc parser stuff.
377 *
378 * Long description. This function has two integer arguments. The first is
379 * @parameter_a and the second is @parameter_b.
380 *
381 * Example: user_function(22);
382 *
383 * Return: Sum of @parameter_a and @parameter_b.
384 *
385 * highlighting:
386 *
387 * - vintage()    : function
388 * - @parameter_a : name of a parameter
389 * - $ENVVAR      : environmental variable
390 * - &my_struct   : name of a structure (up to two words including ``struct``)
391 * - %CONST       : name of a constant.
392 *
393 * Parser Mode: *vintage* kernel-doc mode
394 *
395 * Within the *vintage kernel-doc mode* ignores any whitespace or inline
396 * markup.
397 *
398 * - Inline markup like *emphasis* or **emphasis strong**
399 * - Literals and/or block indent:
400 *
401 *      a + b
402 *
403 * In kernel-doc *vintage* mode, there are no special block or inline markups
404 * available. Markups like the one above result in ambiguous reST markup which
405 * could produce error messages in the subsequently sphinx-build
406 * process. Unexpected outputs are mostly the result.
407 *
408 * This is a link https://git.kernel.org/cgit/linux/kernel/git/torvalds/linux.git/
409 * to the Linux kernel source tree
410 *
411 * colon markup: sectioning by colon markup in vintage mode is partial ugly. ;-)
412 *
413 */

416  return a + b;
